  <div class="garden-footer">风在，树在，小乖在。</div>

<script>
const canvas = document.getElementById('garden');
const ctx = canvas.getContext('2d');
ctx.imageSmoothingEnabled = false;

const W = 480, H = 320;
const S = 2; // pixel scale

// Colors
const SKY = '#d4eaf7';
const SKY2 = '#e8f4fb';
const GROUND = '#8fbc6a';
const GROUND2 = '#7aa855';
const DIRT = '#c4956a';
const DIRT2 = '#b07850';

// Wind state
let windTime = 0;
let windStrength = 0;
let windTarget = 0.3;
let windTimer = 0;

// Wind particles
const windParticles = [];
for (let i = 0; i < 18; i++) {
  windParticles.push({
    x: Math.random() * W,
    y: 80 + Math.random() * 160,
    speed: 0.5 + Math.random() * 1.5,
    size: 1 + Math.floor(Math.random() * 2),
    alpha: 0.1 + Math.random() * 0.3,
    life: Math.random()
  });
}

// Flowers
const flowers = [
  { x: 60, y: 230, type: 0, phase: 0 },
  { x: 90, y: 238, type: 1, phase: 1 },
  { x: 115, y: 233, type: 2, phase: 0.5 },
  { x: 340, y: 235, type: 1, phase: 0.3 },
  { x: 370, y: 228, type: 0, phase: 0.7 },
  { x: 400, y: 240, type: 2, phase: 0.2 },
  { x: 420, y: 232, type: 1, phase: 0.9 },
  { x: 180, y: 240, type: 0, phase: 0.4 },
];

// Small grass tufts
const grasses = [];
for (let i = 0; i < 30; i++) {
  grasses.push({
    x: 20 + Math.random() * 440,
    y: 242 + Math.floor(Math.random() * 15),
    h: 4 + Math.floor(Math.random() * 6),
    phase: Math.random() * Math.PI * 2
  });
}

// Cat state (小乖)
const cat = {
  x: 300,
  y: 220,
  state: 'sit', // sit, walk, tail
  frame: 0,
  frameTimer: 0,
  dir: -1,
  walkTimer: 0,
  stateTimer: 0,
  eyeBlink: 0,
  blinkTimer: 0,
  tailPhase: 0
};

// Tree sway
let treePhase = 0;

function drawPixel(x, y, color, size = S) {
  ctx.fillStyle = color;
  ctx.fillRect(Math.floor(x) * 1, Math.floor(y) * 1, size, size);
}

function drawRect(x, y, w, h, color) {
  ctx.fillStyle = color;
  ctx.fillRect(x, y, w, h);
}

// Draw sky gradient
function drawSky() {
  const grad = ctx.createLinearGradient(0, 0, 0, 200);
  grad.addColorStop(0, '#b8d8f0');
  grad.addColorStop(1, '#e8f4fb');
  ctx.fillStyle = grad;
  ctx.fillRect(0, 0, W, 250);
}

// Draw ground
function drawGround() {
  // Main ground
  ctx.fillStyle = GROUND;
  ctx.fillRect(0, 245, W, H - 245);
  // Ground line
  ctx.fillStyle = GROUND2;
  ctx.fillRect(0, 245, W, 4);
  // Dirt patches
  ctx.fillStyle = DIRT;
  for (let i = 0; i < 6; i++) {
    ctx.fillRect(40 + i * 70, 258, 30, 8);
  }
}

// Draw pixel tree
function drawTree(windSway) {
  const tx = 220, ty = 160;
  const sway = windSway * 3;

  // Trunk
  ctx.fillStyle = '#6b4423';
  ctx.fillRect(tx - 6, ty + 40, 12, 60);
  ctx.fillStyle = '#8b5e3c';
  ctx.fillRect(tx - 4, ty + 40, 4, 60);

  // Roots
  ctx.fillStyle = '#5a3618';
  ctx.fillRect(tx - 14, ty + 90, 10, 6);
  ctx.fillRect(tx + 6, ty + 92, 10, 5);

  // Branches (sway with wind)
  ctx.save();
  ctx.translate(tx, ty + 45);

  // Leaf layers - dark base
  const leafColors = ['#3a7a2a', '#4a9a3a', '#5ab045', '#6bc855'];

  // Bottom layer
  ctx.fillStyle = leafColors[0];
  ctx.beginPath();
  // pixel-style blob
  for (let dy = 0; dy < 5; dy++) {
    const w = [16, 24, 28, 24, 16][dy];
    const xOff = sway * (dy * 0.15);
    ctx.fillRect(-w/2 + xOff, dy * 8 - 10, w, 8);
  }
  ctx.fill();

  // Mid layer
  ctx.fillStyle = leafColors[1];
  for (let dy = 0; dy < 5; dy++) {
    const w = [12, 20, 26, 22, 14][dy];
    const xOff = sway * (dy * 0.2 + 0.1);
    ctx.fillRect(-w/2 + xOff, dy * 8 - 28, w, 8);
  }

  // Top layer
  ctx.fillStyle = leafColors[2];
  for (let dy = 0; dy < 4; dy++) {
    const w = [8, 16, 18, 10][dy];
    const xOff = sway * (dy * 0.25 + 0.2);
    ctx.fillRect(-w/2 + xOff, dy * 8 - 44, w, 8);
  }

  // Highlight
  ctx.fillStyle = leafColors[3];
  for (let dy = 0; dy < 3; dy++) {
    const w = [4, 8, 6][dy];
    const xOff = sway * (dy * 0.3 + 0.3) - 4;
    ctx.fillRect(-w/2 + xOff, dy * 8 - 56, w, 6);
  }

  ctx.restore();
}

// Draw flower
function drawFlower(x, y, type, phase, time) {
  const bob = Math.sin(time * 1.5 + phase) * 1.5;
  const sway = Math.sin(time * 1.2 + phase) * windStrength * 2;

  // Stem
  ctx.fillStyle = '#5a9a3a';
  ctx.fillRect(x - 1, y + bob - 12 + sway * 0.3, 2, 14);

  // Leaves
  ctx.fillStyle = '#4a8a2a';
  ctx.fillRect(x - 5, y + bob - 6 + sway * 0.2, 4, 3);
  ctx.fillRect(x + 2, y + bob - 4 + sway * 0.2, 4, 3);

  const px = x + sway * 0.5;
  const py = y + bob - 14;

  if (type === 0) {
    // Pink flower
    ctx.fillStyle = '#f48fb1';
    ctx.fillRect(px - 4, py - 2, 4, 8);
    ctx.fillRect(px, py - 2, 4, 8);
    ctx.fillRect(px - 2, py - 4, 4, 4);
    ctx.fillRect(px - 2, py + 4, 4, 4);
    ctx.fillStyle = '#ffeb82';
    ctx.fillRect(px - 2, py, 4, 4);
  } else if (type === 1) {
    // White/lavender flower
    ctx.fillStyle = '#d4b8e8';
    for (let a = 0; a < 6; a++) {
      const ax = Math.cos(a * Math.PI / 3) * 4;
      const ay = Math.sin(a * Math.PI / 3) * 4;
      ctx.fillRect(px + ax - 1, py + ay - 1, 3, 3);
    }
    ctx.fillStyle = '#fff8b0';
    ctx.fillRect(px - 2, py - 1, 4, 3);
  } else {
    // Yellow flower
    ctx.fillStyle = '#ffd54f';
    ctx.fillRect(px - 5, py, 4, 4);
    ctx.fillRect(px + 1, py, 4, 4);
    ctx.fillRect(px - 2, py - 4, 4, 4);
    ctx.fillRect(px - 2, py + 4, 4, 4);
    ctx.fillStyle = '#ff8a65';
    ctx.fillRect(px - 2, py, 4, 4);
  }
}

// Draw grass tuft
function drawGrass(g, time) {
  const sway = Math.sin(time * 2 + g.phase) * windStrength * 2;
  ctx.fillStyle = '#5aaa40';
  ctx.fillRect(g.x, g.y, 2, -g.h);
  ctx.fillStyle = '#6aba50';
  ctx.fillRect(g.x + sway, g.y - g.h, 2, -3);
  ctx.fillRect(g.x - 2 - sway * 0.5, g.y - g.h + 2, 2, -2);
}

// Draw pixel cat (小乖 - silver tabby)
function drawCat(cat, time) {
  const x = Math.floor(cat.x);
  const y = Math.floor(cat.y);
  const dir = cat.dir; // 1 = right, -1 = left

  ctx.save();
  if (dir === 1) {
    ctx.translate(x + 12, y);
    ctx.scale(-1, 1);
  } else {
    ctx.translate(x, y);
  }

  const tailSway = Math.sin(time * 2 + cat.tailPhase) * 4;

  if (cat.state === 'sit' || cat.state === 'tail') {
    // Sitting cat body
    // Tail
    ctx.fillStyle = '#9aabba';
    ctx.fillRect(12, 0, 3, 10);
    ctx.fillRect(13 + tailSway, -6, 3, 8);
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(13 + tailSway * 0.7, -10, 3, 5);
    ctx.fillRect(14 + tailSway * 0.5, -14, 3, 5);

    // Body
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(2, -10, 10, 12);
    ctx.fillStyle = '#ccd8e0';
    ctx.fillRect(3, -9, 7, 9);

    // Chest / belly
    ctx.fillStyle = '#e8eef2';
    ctx.fillRect(3, -4, 5, 6);

    // Front paws
    ctx.fillStyle = '#9aabba';
    ctx.fillRect(2, 2, 4, 3);
    ctx.fillRect(8, 2, 4, 3);
    ctx.fillStyle = '#f0f0f0';
    ctx.fillRect(3, 3, 2, 2);
    ctx.fillRect(9, 3, 2, 2);

    // Head
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(1, -20, 12, 12);
    ctx.fillStyle = '#ccd8e0';
    ctx.fillRect(2, -19, 9, 10);

    // Ears
    ctx.fillStyle = '#9aabba';
    ctx.fillRect(1, -24, 4, 5);
    ctx.fillRect(9, -24, 4, 5);
    ctx.fillStyle = '#f4b8c0';
    ctx.fillRect(2, -23, 2, 3);
    ctx.fillRect(10, -23, 2, 3);

    // Eyes
    if (cat.eyeBlink > 0) {
      ctx.fillStyle = '#6a8090';
      ctx.fillRect(3, -16, 4, 1);
      ctx.fillRect(8, -16, 4, 1);
    } else {
      ctx.fillStyle = '#2a5060';
      ctx.fillRect(3, -17, 4, 4);
      ctx.fillRect(8, -17, 4, 4);
      ctx.fillStyle = '#1a3040';
      ctx.fillRect(4, -16, 2, 2);
      ctx.fillRect(9, -16, 2, 2);
      ctx.fillStyle = '#ffffff';
      ctx.fillRect(3, -17, 2, 1);
      ctx.fillRect(8, -17, 2, 1);
    }

    // Nose
    ctx.fillStyle = '#f08080';
    ctx.fillRect(6, -14, 2, 2);
    // Mouth
    ctx.fillStyle = '#8a8a8a';
    ctx.fillRect(5, -12, 2, 1);
    ctx.fillRect(7, -12, 2, 1);

    // Stripes
    ctx.fillStyle = '#8a9dac';
    ctx.fillRect(3, -8, 2, 1);
    ctx.fillRect(3, -5, 2, 1);
    ctx.fillRect(10, -8, 2, 1);
    ctx.fillRect(5, -18, 1, 3);
    ctx.fillRect(8, -18, 1, 3);
    ctx.fillRect(3, -19, 4, 1);

  } else if (cat.state === 'walk') {
    const legFrame = cat.frame % 2;

    // Tail (up and wavy when walking)
    ctx.fillStyle = '#9aabba';
    ctx.fillRect(12, -8, 3, 8);
    ctx.fillRect(11 + tailSway, -16, 3, 10);
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(10 + tailSway * 0.7, -20, 3, 5);

    // Body
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(2, -14, 10, 10);
    ctx.fillStyle = '#ccd8e0';
    ctx.fillRect(3, -13, 7, 8);
    ctx.fillStyle = '#e8eef2';
    ctx.fillRect(4, -8, 4, 5);

    // Legs
    ctx.fillStyle = '#9aabba';
    if (legFrame === 0) {
      ctx.fillRect(2, -2, 3, 5);
      ctx.fillRect(9, -4, 3, 5);
    } else {
      ctx.fillRect(2, -4, 3, 5);
      ctx.fillRect(9, -2, 3, 5);
    }

    // Head
    ctx.fillStyle = '#b8c8d4';
    ctx.fillRect(0, -22, 12, 11);
    ctx.fillStyle = '#ccd8e0';
    ctx.fillRect(1, -21, 9, 9);

    // Ears
    ctx.fillStyle = '#9aabba';
    ctx.fillRect(0, -26, 4, 5);
    ctx.fillRect(8, -26, 4, 5);
    ctx.fillStyle = '#f4b8c0';
    ctx.fillRect(1, -25, 2, 3);
    ctx.fillRect(9, -25, 2, 3);

    // Eyes (determined look when walking)
    if (cat.eyeBlink > 0) {
      ctx.fillStyle = '#6a8090';
      ctx.fillRect(2, -18, 4, 1);
      ctx.fillRect(7, -18, 4, 1);
    } else {
      ctx.fillStyle = '#2a5060';
      ctx.fillRect(2, -19, 4, 3);
      ctx.fillRect(7, -19, 4, 3);
      ctx.fillStyle = '#1a3040';
      ctx.fillRect(3, -18, 2, 2);
      ctx.fillRect(8, -18, 2, 2);
    }

    // Nose
    ctx.fillStyle = '#f08080';
    ctx.fillRect(5, -16, 2, 2);

    // Stripes
    ctx.fillStyle = '#8a9dac';
    ctx.fillRect(3, -11, 2, 1);
    ctx.fillRect(3, -8, 2, 1);
    ctx.fillRect(10, -11, 2, 1);
  }

  ctx.restore();
}

// Draw wind particles
function drawWind(time) {
  windParticles.forEach(p => {
    const alpha = p.alpha * windStrength * 2;
    if (alpha < 0.02) return;
    ctx.fillStyle = `rgba(200, 220, 240, ${alpha})`;
    ctx.fillRect(Math.floor(p.x), Math.floor(p.y), p.size * 2, p.size);
  });
}

// Draw small clouds
function drawClouds(time) {
  const c1x = ((time * 8) % (W + 100)) - 50;
  const c2x = ((time * 5 + 200) % (W + 120)) - 60;

  ctx.fillStyle = 'rgba(255,255,255,0.85)';
  // Cloud 1
  ctx.fillRect(c1x, 30, 40, 12);
  ctx.fillRect(c1x + 6, 22, 28, 14);
  ctx.fillRect(c1x + 14, 16, 16, 10);

  // Cloud 2 (smaller)
  ctx.fillStyle = 'rgba(255,255,255,0.7)';
  ctx.fillRect(c2x, 55, 28, 10);
  ctx.fillRect(c2x + 4, 48, 20, 10);
  ctx.fillRect(c2x + 8, 44, 12, 8);
}

// Draw small birds (pixel)
function drawBirds(time) {
  const b1x = ((time * 15) % (W + 60)) - 30;
  const b1y = 50 + Math.sin(time * 3) * 5;
  const b2x = ((time * 12 + 180) % (W + 60)) - 30;
  const b2y = 65 + Math.sin(time * 2.5 + 1) * 4;

  ctx.fillStyle = '#5a7890';
  // Bird 1
  const wingUp = Math.sin(time * 8) > 0;
  if (wingUp) {
    ctx.fillRect(b1x, b1y, 4, 2);
    ctx.fillRect(b1x + 4, b1y - 2, 2, 2);
    ctx.fillRect(b1x + 6, b1y, 4, 2);
  } else {
    ctx.fillRect(b1x, b1y + 1, 4, 2);
    ctx.fillRect(b1x + 4, b1y, 2, 2);
    ctx.fillRect(b1x + 6, b1y + 1, 4, 2);
  }

  // Bird 2 (smaller, further)
  ctx.fillStyle = '#7a9ab0';
  ctx.fillRect(b2x, b2y, 3, 1);
  ctx.fillRect(b2x + 3, b2y - 1, 2, 2);
  ctx.fillRect(b2x + 5, b2y, 3, 1);
}

// Update cat
function updateCat(dt, time) {
  cat.frameTimer += dt;
  cat.stateTimer += dt;
  cat.blinkTimer += dt;
  cat.tailPhase += dt * 1.5;

  // Blink
  if (cat.blinkTimer > 3 + Math.random() * 2) {
    cat.eyeBlink = 0.15;
    cat.blinkTimer = 0;
  }
  if (cat.eyeBlink > 0) {
    cat.eyeBlink -= dt;
    if (cat.eyeBlink < 0) cat.eyeBlink = 0;
  }

  // Frame
  if (cat.frameTimer > 0.2) {
    cat.frame++;
    cat.frameTimer = 0;
  }

  // State machine
  if (cat.state === 'sit') {
    if (cat.stateTimer > 4 + Math.random() * 3) {
      cat.state = 'walk';
      cat.stateTimer = 0;
      cat.dir = Math.random() > 0.5 ? 1 : -1;
    }
  } else if (cat.state === 'walk') {
    cat.x += cat.dir * 0.5;
    if (cat.x < 30) { cat.dir = 1; }
    if (cat.x > W - 50) { cat.dir = -1; }
    if (cat.stateTimer > 3 + Math.random() * 2) {
      cat.state = Math.random() > 0.3 ? 'sit' : 'tail';
      cat.stateTimer = 0;
    }
  } else if (cat.state === 'tail') {
    if (cat.stateTimer > 2 + Math.random() * 2) {
      cat.state = 'sit';
      cat.stateTimer = 0;
    }
  }
}

// Update wind
function updateWind(dt) {
  windTimer += dt;
  if (windTimer > 2 + Math.random() * 3) {
    windTarget = 0.1 + Math.random() * 0.8;
    windTimer = 0;
  }
  windStrength += (windTarget - windStrength) * dt * 0.5;

  windParticles.forEach(p => {
    p.x += p.speed * windStrength * 3;
    p.life += dt;
    if (p.x > W + 10) {
      p.x = -10;
      p.y = 80 + Math.random() * 160;
      p.alpha = 0.1 + Math.random() * 0.3;
    }
  });
}

let lastTime = null;

function draw(timestamp) {
  if (!lastTime) lastTime = timestamp;
  const dt = Math.min((timestamp - lastTime) / 1000, 0.05);
  lastTime = timestamp;

  const time = timestamp / 1000;
  windTime = time;

  updateWind(dt);
  updateCat(dt, time);

  // Clear
  ctx.clearRect(0, 0, W, H);

  // Sky
  drawSky();

  // Clouds
  drawClouds(time);

  // Birds
  drawBirds(time);

  // Wind particles (behind tree)
  drawWind(time);

  // Tree
  const treeSway = Math.sin(time * 1.8) * windStrength + Math.sin(time * 2.7) * windStrength * 0.4;
  drawTree(treeSway);

  // Ground
  drawGround();

  // Grasses
  grasses.forEach(g => drawGrass(g, time));

  // Flowers
  flowers.forEach(f => drawFlower(f.x, f.y, f.type, f.phase, time));

  // Cat
  drawCat(cat, time);

  requestAnimationFrame(draw);
}

requestAnimationFrame(draw);
